\begin{align} -\infty \amp \lt (b-a)\inf_{[a,b]} f = \sum_{n=1}^N (x_n-x_{n-1}) \inf_{ [a,b]} f\notag\\ \amp \leq \sum_{n=1}^N (x_n-x_{n-1}) \inf_{ [x_{n-1}, x_n]} f = L(f; x_0, \dots, x_N)\notag\\ \amp\leq \sum_{n=1}^N (x_n-x_{n-1}) \sup_{ [x_{n-1}, x_n]} f = U(f; x_0, \dots, x_N)\notag\\ \amp \leq \sum_{n=1}^N (x_n-x_{n-1}) \sup_{ [a,b]} f \notag\\ \amp = (b-a)\sup_{[a,b]} f\lt \infty.\tag{9.1} \end{align}
